    Include /etc/proftpd/modules.conf
    
    ## サーバー名とバージョン情報
    ServerName  "ftp.avswa.com"  ## アクセス時に表示されるサーバー名
    ServerIdent  no              ## ソフトウェアバージョンを隠す
    ServerAdmin  root@localhost
    DefaultServer  on
    
    ## 上位階層への移動禁止(chroot)
    DefaultRoot  ~
    
    ## 認証機構(PAM)
    AuthPAMConfig  proftpd
    AuthOrder  mod_auth_pam.c* mod_auth_unix.c
    
    ## 逆引き(DNS)
    UseReverseDNS  off
    
    ## 実行ユーザーと実行グループ
    User  nobody
    Group  nobody
    
    ## DOS攻撃対処
    MaxInstances  20
    
    ## SendFile機能を無効化
    UseSendfile  off
    
    ## ログフォーマット定義
    LogFormat  default  "%h %l %u %t \"%r\" %s %b"
    LogFormat  auth  "%v [%P] %h %t \"%r\" %s"
    
    ## ログオプション
    LogOptions  -Timestamp -Hostname +RoleBasedProcessLabels
    
    ## 基本設定(制御プロセス)
    ControlsEngine  on
    ControlsACLs  all allow user root
    ControlsSocketACL  allow user *
    ControlsLog  /var/log/proftpd/controls.log
    
    ## 追加設定(制御プロセス)
    <IfModule mod_ctrls_admin.c>
      AdminControlsEngine  on
      AdminControlsACLs  all allow user root
    </IfModule>
    
    ## 追加設定(制御プロセス)
    <IfModule mod_vroot.c>
      VRootEngine  on
    </IfModule>
    
    ## SSL/TLS関連
    <IfDefine TLS>
      Include /etc/proftpd/mod_tls.conf
    </IfDefine>
    
    ## BAN関連
    <IfModule mod_ban.c>
      Include /etc/proftpd/mod_ban.conf
    </IfModule>
    
    ## QOS関連
    <IfModule mod_qos.c>
      Include /etc/proftpd/mod_qos.conf
    </IfModule>
    
    ## 操作権限
    <Global>
      Umask  022
      AllowOverwrite  yes
      <Limit ALL SITE_CHMOD>
        AllowAll
      </Limit>
    </Global>
    
    ## Anonymous接続
    <IfDefine ANONYMOUS_FTP>
      Include /etc/proftpd/anonftp.conf
    </IfDefine>
    
    ## 追加設定ファイル
    Include /etc/proftpd/conf.d/*.conf
