    コマンドの実行結果
    --------------------------------------------------
    // 単にmakeと実行した場合は一番上のターゲットが実行
    // ファイル行頭の空白はTab
    
    PROG := hello
    OBJ1 := main.o
    OBJ2 := hello.o
    SRC1 := main.c
    SRC2 := hello.c
    COMP := gcc
    
    $(PROG): $(OBJ1) $(OBJ2)
      $(COMP) $(OBJ1) $(OBJ2) -o $(PROG)
    
    $(OBJ1): $(SRC1)
      $(COMP) -c $(SRC1)
    
    $(OBJ2): $(SRC2)
      $(COMP) -c $(SRC2)
    
    clean:
      rm -f $(PROG) $(OBJ1) $(OBJ2)
    --------------------------------------------------
