  File System - 1 (設定ファイルの書式)
  
[内容]
  ## 設定ファイル(/etc/fstab)の書式
  --------------------------------------------------
	デバイス名(UUID名)/ラベル名  マウントポイント  ファイルシステム種類  マウントオプション  dump対象  fsck順序
	
	(設定例)
	/dev/sda1  /boot  ext2  defaults  0  2
	
	ファイルシステム種類
		ext2				// Linux用ファイルシステム「ext」の拡張版
		ext3				// ext2へジャーナリング機能付加
		ext4				// ext3を機能拡張
		ReiserFS		// Linux用ジャーナリングファイルシステム
		JFS					// IBM製ジャーナリングファイルシステム
		XFS					// SGI製ジャーナリングファイルシステム → CentOS7のデフォルト
		Btrfs				// 先進的ファイルシステム
		ISO9660		// CD用
		UDF					// DVD用
		F2FS				// SSD用
		msdos				// MS-DOS用
		FAT/FAT32	// フラッシュメモリ用
		NTFS				// WindowsNT/2000以降
		hpfs				// OS/2用
		HFS					// MacOS用
		HFS+				// MacOS8.1以降
		NFS					// ネットワークファイルシステム
		CIFS				// Windowsファイル共有用
		procfs			// プロセス情報を扱う仮想ファイルシステム(/proc)
		sysfs				// デバイス情報を扱う仮想ファイルシステム
		tmpfs				// 仮想メモリを扱う仮想ファイルシステム
		devpts			// 疑似端末制御のための仮想ファイルシステム
		usbfs				// USBデバイス監視用
		cramfs			// 組込システム用
	
	マウントオプション
		--------------------------------------------------
		入出力同期
			async				// 非同期
			sync				// 同期
		--------------------------------------------------
		iノードアクセス時刻更新
			atime				// 更新
			relatime		// 一定時間後に更新
			noatime		// 更新しない → ディスクアクセス高速化
		--------------------------------------------------
		コマンド「mount -a」
			auto				// マウント
			noauto			// マウントしない
		--------------------------------------------------
		バイナリ実行
			exec				// バイナリ実行を許可
			noexec			// バイナリ実行を禁止
		--------------------------------------------------
		SUID, SGID
			suid				// 有効
			nosuid			// 無効
		--------------------------------------------------
		読取専用
			ro					// 読取専用でマウント
			rw					// 読書許可でマウント
		--------------------------------------------------
		一般ユーザーのマウント
			user				// 一般ユーザーマウント許可、アンマウントはマウント操作したユーザーのみ
			users				// 一般ユーザーマウント許可、アンマウントもマウント操作したユーザー以外で許可
			nouser			// 一般ユーザーマウント禁止
		--------------------------------------------------
		その他
			owner				// デバイスファイル所有者のマウント許可
			defaults		// async, auto, dev, exec, nouser, rw, suid
			dev					// ファイルシステム上のデバイスファイル使用可
			group				// 条件「ユーザーのグループ = デバイスファイルの所有グループ」が満たされた場合にマウント許可
			uid=UID		// 全ファイル所有者を指定UIDにする
			gid=GID		// 全ファイル所有グループを指定GIDにする
	
	dump対象
		0						// コマンド「dump」のバックアップ対象外
		0以外				// コマンド「dump」でバックアップ
	
	fsck順序
		0						// システム起動時のコマンド「fsck」のチェック対象外
		0以外				// システム起動時のコマンド「fsck」のチェック順序(1, 2‥の順)
  --------------------------------------------------
  
[確認]
